1    package com.github.smuddgge.events;
2    
3    import com.github.smuddgge.managers.ServerThreadNetworkManager;
4    import com.github.smuddgge.packets.Packet;
5    import com.github.smuddgge.server.Server;
6    import com.github.smuddgge.utility.GameRoom;
7    
8    /**
9     * Event used to delete the game room they are in
10    */
11   public class GameRoomDeleteEvent extends Event {
12   
13       @Override
14       public void update(Packet credentials, Server server, ServerThreadNetworkManager serverThreadNetworkManager) {
15           // Check if the game room exists, and they are in the game room
16           GameRoom gameRoom = server.getGameRoom(serverThreadNetworkManager.getPlayerProfile());
17           if (gameRoom == null) return;
18           if (!gameRoom.containsPlayer(serverThreadNetworkManager.getPlayerProfile().uuid)) return;
19   
20           // Remove the game room
21           server.removeGameRoom(gameRoom);
22       }
23   }
